```python
class ExamPaper(Base):
    __tablename__ = "exam_papers"

    id = Column(UUID(as_uuid=True), primary_key=True, default=uuid.uuid4)
    title = Column(Text, nullable=False)
    level = Column(String(5), nullable=False)  # N1|N2|N3|N4|N5
    source = Column(Text, nullable=True)
    created_at = Column(DateTime(timezone=True), nullable=False, server_default=text("now()"))

    sections = relationship("ExamSection", back_populates="paper", cascade="all, delete-orphan",
                            order_by="ExamSection.seq")

    __table_args__ = (
        UniqueConstraint("title", "level", name="uq_exam_papers_title_level"),
        Index("ix_exam_papers_level", "level"),
    )


class ExamSection(Base):
    __tablename__ = "exam_sections"

    id = Column(UUID(as_uuid=True), primary_key=True, default=uuid.uuid4)
    paper_id = Column(UUID(as_uuid=True), ForeignKey("exam_papers.id", ondelete="CASCADE"), nullable=False)
    name = Column(String(50), nullable=False)   # 文法|読解|聴解|文字・語彙
    seq = Column(Integer, nullable=False)
    created_at = Column(DateTime(timezone=True), nullable=False, server_default=text("now()"))

    paper = relationship("ExamPaper", back_populates="sections")
    questions = relationship("ExamQuestion", back_populates="section", cascade="all, delete-orphan",
                             order_by="ExamQuestion.seq")

    __table_args__ = (
        Index("ix_exam_sections_paper_id", "paper_id"),
    )


class ExamQuestion(Base):
    __tablename__ = "exam_questions"

    id = Column(UUID(as_uuid=True), primary_key=True, default=uuid.uuid4)
    section_id = Column(UUID(as_uuid=True), ForeignKey("exam_sections.id", ondelete="CASCADE"), nullable=False)
    type = Column(String(20), nullable=False)   # grammar|reading|listening|ordering
    stem = Column(Text, nullable=False)
    options = Column(JSONB, nullable=False)     # {"A": "のに", "B": "くせに", ...}
    correct_answer = Column(String(1), nullable=False)  # A|B|C|D
    seq = Column(Integer, nullable=False)
    created_at = Column(DateTime(timezone=True), nullable=False, server_default=text("now()"))

    section = relationship("ExamSection", back_populates="questions")
    analysis = relationship("QuestionAnalysis", back_populates="question", uselist=False,
                            cascade="all, delete-orphan")

    __table_args__ = (
        Index("ix_exam_questions_section_id", "section_id"),
    )


class QuestionAnalysis(Base):
    __tablename__ = "question_analyses"

    id = Column(UUID(as_uuid=True), primary_key=True, default=uuid.uuid4)
    question_id = Column(UUID(as_uuid=True), ForeignKey("exam_questions.id", ondelete="CASCADE"),
                         nullable=False, unique=True)
    session_data = Column(JSONB, nullable=True)         # AI 分析结果 + 追问历史
    relations_suggested = Column(JSONB, nullable=True)  # 语法关系建议列表
    created_at = Column(DateTime(timezone=True), nullable=False, server_default=text("now()"))

    question = relationship("ExamQuestion", back_populates="analysis")


class ExamAttempt(Base):
    __tablename__ = "exam_attempts"

    id = Column(UUID(as_uuid=True), primary_key=True, default=uuid.uuid4)
    paper_id = Column(UUID(as_uuid=True), ForeignKey("exam_papers.id", ondelete="CASCADE"), nullable=False)
    status = Column(String(20), nullable=False, server_default=text("'in_progress'"))
    score = Column(JSONB, nullable=True)  # {"文法": {"correct": 8, "total": 13}, ...}
    started_at = Column(DateTime(timezone=True), nullable=False, server_default=text("now()"))
    completed_at = Column(DateTime(timezone=True), nullable=True)

    paper = relationship("ExamPaper")
    answers = relationship("AttemptAnswer", back_populates="attempt", cascade="all, delete-orphan")

    __table_args__ = (
        Index("ix_exam_attempts_paper_id", "paper_id"),
        Index("ix_exam_attempts_status", "status"),
    )


class AttemptAnswer(Base):
    __tablename__ = "attempt_answers"

    attempt_id = Column(UUID(as_uuid=True), ForeignKey("exam_attempts.id", ondelete="CASCADE"),
                        primary_key=True)
    question_id = Column(UUID(as_uuid=True), ForeignKey("exam_questions.id", ondelete="CASCADE"),
                         primary_key=True)
    user_answer = Column(String(1), nullable=False)  # A|B|C|D
    is_correct = Column(Boolean, nullable=False)

    attempt = relationship("ExamAttempt", back_populates="answers")
    question = relationship("ExamQuestion")

    __table_args__ = (
        Index("ix_attempt_answers_attempt_id", "attempt_id"),
    )
```

```python
from uuid import UUID
from datetime import datetime
from pydantic import BaseModel


class SectionSummary(BaseModel):
    id: UUID
    name: str
    seq: int
    question_count: int


class QuestionOption(BaseModel):
    A: str
    B: str
    C: str
    D: str


class QuestionItem(BaseModel):
    id: UUID
    type: str
    stem: str
    options: QuestionOption
    seq: int
    # correct_answer 不在列表接口里返回，防止作弊


class QuestionWithAnswer(BaseModel):
    id: UUID
    type: str
    stem: str
    options: QuestionOption
    seq: int
    correct_answer: str  # 提交后才返回


class SectionDetail(BaseModel):
    id: UUID
    name: str
    seq: int
    questions: list[QuestionItem]


class ExamPaperList(BaseModel):
    id: UUID
    title: str
    level: str
    source: str | None
    section_count: int
    question_count: int
    created_at: datetime

    model_config = {"from_attributes": True}


class ExamPaperDetail(BaseModel):
    id: UUID
    title: str
    level: str
    source: str | None
    sections: list[SectionDetail]
    created_at: datetime


class StartAttemptResponse(BaseModel):
    attempt_id: UUID
    paper_id: UUID
    status: str


class SubmitAnswerRequest(BaseModel):
    question_id: UUID
    answer: str  # A|B|C|D


class SubmitAnswerResponse(BaseModel):
    question_id: UUID
    recorded: bool


class SectionScore(BaseModel):
    correct: int
    total: int


class SubmitSectionResponse(BaseModel):
    section_name: str
    score: SectionScore
    answers: list[dict]  # [{question_id, user_answer, is_correct, correct_answer}]


class AttemptStatus(BaseModel):
    attempt_id: UUID
    paper_id: UUID
    status: str
    score: dict | None
    answered_question_ids: list[UUID]


class RelationSuggestion(BaseModel):
    from_key: str
    to_key: str
    type: str
    note: str


class QuestionAnalysisResponse(BaseModel):
    question_id: UUID
    session_data: dict | None
    relations_suggested: list[RelationSuggestion]
    cached: bool
```

```python
@router.get("/exams", response_model=list[ExamPaperList])
async def list_exams(db: AsyncSession = Depends(get_db)):
    result = await db.execute(
        select(
            ExamPaper,
            func.count(ExamSection.id.distinct()).label("section_count"),
            func.count(ExamQuestion.id).label("question_count"),
        )
        .outerjoin(ExamSection, ExamSection.paper_id == ExamPaper.id)
        .outerjoin(ExamQuestion, ExamQuestion.section_id == ExamSection.id)
        .group_by(ExamPaper.id)
        .order_by(ExamPaper.created_at.desc())
    )
    rows = result.all()
    out = []
    for paper, sc, qc in rows:
        out.append(ExamPaperList(
            id=paper.id,
            title=paper.title,
            level=paper.level,
            source=paper.source,
            section_count=sc,
            question_count=qc,
            created_at=paper.created_at,
        ))
    return out

@router.put("/attempts/{attempt_id}/answers", response_model=SubmitAnswerResponse)
async def submit_answer(
    attempt_id: UUID,
    req: SubmitAnswerRequest,
    db: AsyncSession = Depends(get_db),
):
    attempt_res = await db.execute(select(ExamAttempt).where(ExamAttempt.id == attempt_id))
    attempt = attempt_res.scalar_one_or_none()
    if attempt is None:
        raise HTTPException(status_code=404, detail="Attempt not found")

    q_res = await db.execute(select(ExamQuestion).where(ExamQuestion.id == req.question_id))
    question = q_res.scalar_one_or_none()
    if question is None:
        raise HTTPException(status_code=404, detail="Question not found")

    is_correct = req.answer.upper() == question.correct_answer.upper()

    existing = await db.execute(
        select(AttemptAnswer).where(
            AttemptAnswer.attempt_id == attempt_id,
            AttemptAnswer.question_id == req.question_id,
        )
    )
    answer_row = existing.scalar_one_or_none()
    if answer_row:
        answer_row.user_answer = req.answer.upper()
        answer_row.is_correct = is_correct
    else:
        answer_row = AttemptAnswer(
            attempt_id=attempt_id,
            question_id=req.question_id,
            user_answer=req.answer.upper(),
            is_correct=is_correct,
        )
        db.add(answer_row)

    await db.commit()
    return SubmitAnswerResponse(question_id=req.question_id, recorded=True)

@router.post("/attempts/{attempt_id}/sections/{section_id}/submit", response_model=SubmitSectionResponse)
async def submit_section(
    attempt_id: UUID,
    section_id: UUID,
    db: AsyncSession = Depends(get_db),
):
    attempt_res = await db.execute(select(ExamAttempt).where(ExamAttempt.id == attempt_id))
    attempt = attempt_res.scalar_one_or_none()
    if attempt is None:
        raise HTTPException(status_code=404, detail="Attempt not found")

    sec_res = await db.execute(select(ExamSection).where(ExamSection.id == section_id))
    section = sec_res.scalar_one_or_none()
    if section is None:
        raise HTTPException(status_code=404, detail="Section not found")

    q_res = await db.execute(
        select(ExamQuestion).where(ExamQuestion.section_id == section_id).order_by(ExamQuestion.seq)
    )
    questions = q_res.scalars().all()
    question_ids = [q.id for q in questions]

    ans_res = await db.execute(
        select(AttemptAnswer).where(
            AttemptAnswer.attempt_id == attempt_id,
            AttemptAnswer.question_id.in_(question_ids),
        )
    )
    answers = {a.question_id: a for a in ans_res.scalars().all()}

    correct_count = sum(1 for a in answers.values() if a.is_correct)
    total = len(questions)

    score = attempt.score or {}
    score[section.name] = {"correct": correct_count, "total": total}
    total_correct = sum(v["correct"] for v in score.values())
    total_questions = sum(v["total"] for v in score.values())
    score["total"] = {"correct": total_correct, "total": total_questions}

    await db.execute(
        update(ExamAttempt).where(ExamAttempt.id == attempt_id).values(score=score)
    )
    await db.commit()

    answer_details = []
    for q in questions:
        a = answers.get(q.id)
        answer_details.append({
            "question_id": str(q.id),
            "user_answer": a.user_answer if a else None,
            "is_correct": a.is_correct if a else False,
            "correct_answer": q.correct_answer,
        })

    return SubmitSectionResponse(
        section_name=section.name,
        score=SectionScore(correct=correct_count, total=total),
        answers=answer_details,
    )


# ── 题目 AI 分析（懒生成 + 缓存） ────────────────────────────────────────────

@router.get("/questions/{question_id}/analysis", response_model=QuestionAnalysisResponse)
async def get_question_analysis(question_id: UUID, db: AsyncSession = Depends(get_db)):
    q_res = await db.execute(select(ExamQuestion).where(ExamQuestion.id == question_id))
    question = q_res.scalar_one_or_none()
    if question is None:
        raise HTTPException(status_code=404, detail="Question not found")

    # 检查缓存
    cache_res = await db.execute(
        select(QuestionAnalysis).where(QuestionAnalysis.question_id == question_id)
    )
    cached = cache_res.scalar_one_or_none()

    if cached and cached.session_data:
        relations = [RelationSuggestion(**r) for r in (cached.relations_suggested or [])]
        return QuestionAnalysisResponse(
            question_id=question_id,
            session_data=cached.session_data,
            relations_suggested=relations,
            cached=True,
        )

    # 生成分析（grammar 类型）
    if question.type != "grammar":
        # 非语法题暂不支持，返回空
        return QuestionAnalysisResponse(
            question_id=question_id,
            session_data=None,
            relations_suggested=[],
            cached=False,
        )

    opts = question.options
    question_text = (
        f"{question.stem}\n"
        f"A. {opts.get('A', '')}\n"
        f"B. {opts.get('B', '')}\n"
        f"C. {opts.get('C', '')}\n"
        f"D. {opts.get('D', '')}"
    )
    schema_json = json.dumps(_EXAM_GRAMMAR_SCHEMA, ensure_ascii=False)
    prompt = EXAM_GRAMMAR_ANALYSIS.format(
        question_text=question_text,
        correct_answer=question.correct_answer,
        schema_json=schema_json,
    )

    llm = get_llm_client()
    try:
        result_data = await llm.analyze(prompt, _EXAM_GRAMMAR_SCHEMA)
    except Exception as e:
        logger.error("LLM analysis failed for question %s: %s", question_id, e)
        raise HTTPException(status_code=502, detail="AI analysis failed")

    relations_raw = result_data.get("relations_suggested", [])

    # 存入缓存
    if cached:
        cached.session_data = result_data
        cached.relations_suggested = relations_raw
    else:
        new_analysis = QuestionAnalysis(
            question_id=question_id,
            session_data=result_data,
            relations_suggested=relations_raw,
        )
        db.add(new_analysis)
    await db.commit()

    relations = [RelationSuggestion(**r) for r in relations_raw]
    return QuestionAnalysisResponse(
        question_id=question_id,
        session_data=result_data,
        relations_suggested=relations,
        cached=False,
    )


# ── 追问（追加到 question_analyses.session_data） ────────────────────────────

@router.post("/questions/{question_id}/analysis/followup")
async def followup_question_analysis(
    question_id: UUID,
    body: dict,
    db: AsyncSession = Depends(get_db),
):
    cache_res = await db.execute(
        select(QuestionAnalysis).where(QuestionAnalysis.question_id == question_id)
    )
    cached = cache_res.scalar_one_or_none()
    if cached is None:
        raise HTTPException(status_code=404, detail="No analysis found. Call GET first.")

    free_prompt = body.get("prompt", "")
    if not free_prompt:
        raise HTTPException(status_code=400, detail="prompt is required")

    llm = get_llm_client()
    response_text = await llm.complete(free_prompt)

    session = cached.session_data or {}
    followups = session.get("followups", [])
    followups.append({"prompt": free_prompt, "response": response_text})
    session["followups"] = followups

    await db.execute(
        update(QuestionAnalysis)
        .where(QuestionAnalysis.question_id == question_id)
        .values(session_data=session)
    )
    await db.commit()
    return {"response": response_text}
```

```tsx
              <span className={[
                'text-xs font-bold px-1.5 py-0.5 rounded',
                opt.is_correct ? 'bg-emerald-500 text-white' : 'bg-border text-fg-subtle',
              ].join(' ')}>
                {opt.option} {opt.is_correct ? '✓' : ''}
              </span>
              <span className="text-fg-subtle text-xs">{opt.grammar?.pattern}</span>
            </div>
            <p className="text-fg leading-snug">{opt.explanation}</p>
          </div>
        ))}
      </div>

      {/* 关系建议 */}
      {data.relations_suggested.length > 0 && (
        <RelationSuggestions relations={data.relations_suggested} />
      )}

      {/* 追问历史 */}
      {followups.map((f, i) => (
        <div key={i} className="flex flex-col gap-1">
          <p className="text-fg-subtle text-xs">你：{f.prompt}</p>
          <p className="text-fg leading-snug">{f.response}</p>
        </div>
      ))}

      {/* 追问输入 */}
      <div className="flex gap-2">
        <input
          value={followup}
          onChange={e => setFollowup(e.target.value)}
          onKeyDown={e => e.key === 'Enter' && !e.shiftKey && handleFollowup()}
          placeholder="继续追问…"
          className="input flex-1 text-sm"
          disabled={asking}
        />
        <button className="btn-primary text-sm" onClick={handleFollowup} disabled={asking || !followup.trim()}>
          {asking ? '…' : '发送'}
        </button>
      </div>
    </div>
  )
}

interface OptionAnalysis {
  option: string
  is_correct: boolean
  explanation: string
  grammar?: { pattern: string }
}
```

```tsx
import { useState } from 'react'
import { getExam } from '../services/api'
import type { ExamPaperDetail, ExamSectionDetail, SubmitSectionResult } from '../types'
import ExamList from '../components/exam/ExamList'
import ExamSession from '../components/exam/ExamSession'
import ExamResults from '../components/exam/ExamResults'

type View = 'list' | 'section-select' | 'session' | 'results'

export default function JlptPage() {
  const [view, setView] = useState<View>('list')
  const [paper, setPaper] = useState<ExamPaperDetail | null>(null)
  const [attemptId, setAttemptId] = useState<string | null>(null)
  const [currentSection, setCurrentSection] = useState<ExamSectionDetail | null>(null)
  const [sectionResult, setSectionResult] = useState<SubmitSectionResult | null>(null)
  const [sectionIdx, setSectionIdx] = useState(0)

  const handleStart = async (paperId: string, aId: string) => {
    const detail = await getExam(paperId)
    setPaper(detail)
    setAttemptId(aId)
    setSectionIdx(0)
    setCurrentSection(detail.sections[0] ?? null)
    setView('session')
  }

  const handleSubmit = (result: SubmitSectionResult) => {
    setSectionResult(result)
    setView('results')
  }

  const handleContinue = () => {
    if (!paper) return
    const nextIdx = sectionIdx + 1
    if (nextIdx < paper.sections.length) {
      setSectionIdx(nextIdx)
      setCurrentSection(paper.sections[nextIdx])
      setView('session')
    } else {
      setView('list')
      setPaper(null)
    }
  }

  return (
    <div className="flex flex-col flex-1 min-h-0 overflow-y-auto p-6 gap-4">
      {/* 标题栏 */}
      <div className="flex items-center gap-3">
        {view !== 'list' && (
          <button className="btn-ghost text-sm" onClick={() => setView('list')}>
            ← 试卷列表
          </button>
        )}
        <h1 className="text-lg font-semibold text-fg">
          {view === 'list' && 'JLPT 专题'}
          {view === 'session' && `${paper?.title} · ${currentSection?.name}`}
          {view === 'results' && `${currentSection?.name} 结果`}
        </h1>
        {view === 'session' && paper && (
          <span className={`badge badge-${paper.level.toLowerCase()} text-xs ml-auto`}>
            {paper.level}
          </span>
        )}
      </div>

      {/* 内容区 */}
      {view === 'list' && (
        <ExamList onStart={handleStart} />
      )}

      {view === 'session' && currentSection && attemptId && (
        <ExamSession
          section={currentSection}
          attemptId={attemptId}
          onSubmit={handleSubmit}
        />
      )}

      {view === 'results' && currentSection && sectionResult && (
        <ExamResults
          section={currentSection}
          result={sectionResult}
          onContinue={handleContinue}
        />
      )}
    </div>
  )
}
```
